\vspace{3.5ex}

\begin{block}{tibble}
  \inline{tibble(x, y)} \\ Create a new tibble with columns x and y.\br
  \inline{as\_tibble(x)} \\ Cast object x into a tibble.\br
  Use as\_tibble(x) if x is a matrix, a list, or a data.frame.
\end{block}

\vspace{4ex}

\begin{block}{readr}
  \blocksubtitle{Reading files}
    \inline{read\_csv(file)} \\ Read a comma-separated file.\br
    \inline{read\_delim(file, delim = ...)} \\ Read a ...-separated file.\br
    Use the option \inline{show_col_types = FALSE} to silence printing of column types.

  \blocksubtitle{Writing files}
    \inline{write\_csv(x, file)}\\ Write table x as a comma-separated file.\br
    \inline{write\_delim(x, file, delim = ...)} \\ Write table x as a ...-separated file.
\end{block}
